// For Google Workspace - Use OAuth (FREE)
class GoogleWorkspaceIntegration(private val userCredential: Credential) {
    fun getEmails(): List<Email> {
        val gmail = Gmail.Builder(transport, factory, userCredential).build()
        return gmail.users().messages().list("me").execute().messages
        // Cost: $0 (uses user's quota)
    }
}

// For Everything Else - Use Composio
class ComposioIntegration(private val apiKey: String) {
    private val composio = Composio(apiKey)
    
    fun getNotionPages(connectedAccountId: String): List<Page> {
        return composio.execute(
            connectedAccountId = connectedAccountId,
            appName = "notion",
            actionName = "notion_list_pages"
        )
        // Cost: Counts against Composio quota
    }
    
    fun createAsanaTask(connectedAccountId: String, task: Task) {
        composio.execute(
            connectedAccountId = connectedAccountId,
            appName = "asana",
            actionName = "asana_create_task",
            params = mapOf("task" to task)
        )
    }
}
